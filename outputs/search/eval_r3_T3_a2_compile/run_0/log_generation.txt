struct node {
	int value;
	node* next;
};
 
void push_back(node** head_ref,int new_data)
{
	node* new_node = new node;
	new_node->value = new_data;
	new_node->next = NULL;
	node* last = *head_ref;
	if (*head_ref == NULL)
	{
		*head_ref = new_node;
		return;
	}
	while (last->next!=NULL)
		last = last->next;
	last->next = new_node;
}
 
void print_list(node* n)
{
	while (n!=NULL)
	{
		cout << n->value << " ";
		n = n->next;
	}
	cout << endl;
}

#include <random>
#include <thread>
#include <tuple>

using namespace std;
template <class T>
struct Mutateable {
  T value;
};

int main() {
  Mutateable<int> mut;
  mut.value = 42;
  thread t( [] { mut.value = 43; });
  auto& gen = random_device {}();
  mt19937 g{gen()};

  uniform_int_distribution<int> d(-1000, 1000);
  t.join();
   
  if(d(g) > 0)
    cout << mut.value << endl;
  else
    cout << -mut.value << endl;
  return 0;
}
