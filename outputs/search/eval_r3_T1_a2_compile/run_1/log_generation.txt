{
    std::list<int> out;
    for(std::vector<int>::const_iterator i = in.begin(); i!= in.end(); ++i)

    public:

            return size_;
        };

        std::size_t size() const;
    private:
        std::size_t size_;
        std::map<std::string, int> data_;

{
    size_ = 0;
}

}

{
    ++data_[val];
    ++size_;
}

{
    return size_;
}
