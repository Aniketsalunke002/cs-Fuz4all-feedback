#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <cstdlib>
#include <thread>
#include <chrono>

#include <random>

using namespace std;
void print(const string & str)
{
    cout << str << endl;
}

string GetStrWithRandomNum(int len)
{
string str;
random_device rd;
default_random_engine gen(rd());
uniform_int_distribution<> dist(0,25);
str.reserve(len);
cout << "Generating random string, length: " << len << endl;
for (int i = 0; i < len; i++) {
    cout << "* ";
    cout.flush();
    str.push_back(dist(gen));
    cout << dist(gen) << endl;
}

return str;
}

void Thread1()
{
print("Thread 1 is started");
print(GetStrWithRandomNum(20));
print("Thread 1 is finished");
}
void Thread2()
{
print("Thread 2 is started");
this_thread::sleep_for(chrono::milliseconds(rand() % 2000 + 2000));
print(GetStrWithRandomNum(20));
print("Thread 2 is finished");
}

int main ()
{
thread t1(Thread1);
thread t2(Thread2);

t1.join();
t2.join();
return 0;
}
