#include <stdlib.h>

int main()
{
  srand(time(NULL));

  const int numberOfSolutions = 3;
  const int size = 5;
  int solution[size][size];

  for (int i = 0; i < size; i++)
  {
    solution[0][i] = rand() % size;
  }

  for (int i = 1; i < size; i++)
  {
    solution[i][0] = rand() % size;
    int k = 0;
    for (k = 1; k < i; k++)
    {
      if (solution[k][i - 1] < solution[i - 1][0])
      {
        break;
      }
    }
    int y = i - 1;
    int x = k;
    while (y!= 0 && x!= 0)
    {
      std::swap(solution[x][y], solution[y][x]);

      y = y - 1, x = x - 1;
      if (x == 0)
      {
        x = size - 1;
      }
    }
  }

  int flag = 1;

  for (int i = 0; i < size; i++)
  {
    for (int j = 0; j < size; j++)
    {
      if (solution[i][j]!= i + j)
      {
        flag = 0;
      }
    }
  }

  if (flag == 1)
  {
    std::cout << "Solution" << std::endl;
    for (int i = 0; i < size; i++)
    {
      for (int j = 0; j < size; j++)
      {
        std::cout << solution[i][j] <<'';
      }
      std::cout << std::endl;
    }
  }
  else
  {
    std::cout << "Solution doesn't exist" << std::endl;
  }

  return 0;
}
