using namespace std;

class Solution {
private:
    unordered_map<long, int> memory;

public:
    int numberOfSubstrings(string s) {
        int res = s.length(), n = s.length();
        memory.clear();
        memory[0] = 0;
        for (int i = 0; i < n; i++) {
            res -= (1 - isCountInMap(s.at(i))) * memory[1 - s.at(i)];
            if (res < 0) {
                res = 0;
                break;
            }
            memory[s.at(i)] ^= 1;
        }
        return res + 1 - isCountInMap(0);
    }

    bool isCountInMap(char number) {
        return (memory.find(number)!= memory.end());
    }
};

    int numbersCount = 0;
    for (int i = 0; i <= number; i++)
    {
        numbers.push_back(i);
        numbersCount = numbers.size();
    }

    int minValue = numbers[0], temp = 0;
    for (int i = 0; i <= numbersCount; i++)
    {
        minValue = numbers[i];
        for (int j = 0; j < i; j++)
        {
            if (minValue > numbers[j])
            {
                minValue = numbers[j];
            }
        }
        if (temp!= minValue)
        {
            numbers[i] = numbers[numbers.size() - 1];
        }
        numbers.pop_back();
        temp = minValue;
    }

    for (int i = 0; i < numbersCount; i++)
    {
        std::cout << numbers[i] << " ";
    }
    return 0;
}

			 length = j;
		}

		length = 0;
	}
	std::cout << "The length of the biggest segment is " << length << "\n";

		std::cout << std::setprecision(2) << std::fixed << std::setw(12) << x[i];
	}

class Circle {
public:
  Circle(float x,float y,float radius): cx(x), cy(y), cr(radius) {}
  void getArea() {
    std::cout << "Area = " << (float) M_PI*cr*cr << std::endl;
  }
  void getCircumference() {
    std::cout << "Circumference = " << (float) 2*M_PI*cr << std::endl;
  }
  void getTouch(float y) {
    std::cout << "Touch: (" << cx << ", " << y << ")" << std::endl;
  }
  float cx, cy, cr;  
};

c.getTouch(10);
