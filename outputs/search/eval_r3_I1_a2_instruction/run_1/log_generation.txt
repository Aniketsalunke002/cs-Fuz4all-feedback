#include <memory>

template<typename T>
class MyList
{

	public:

		class Node
		{
			friend class MyList<T>;

			bool   isTail;
			T      data;
			std::shared_ptr<Node> head;
			std::shared_ptr<Node> tail;

		};

		std::shared_ptr<Node> head;
		std::shared_ptr<Node> tail;

		MyList() : head(new Node){ head->isTail = true;};
		
		void add (T d)
		{
			std::shared_ptr<Node> ptr = std::shared_ptr<Node>(new Node);
			ptr->data = d;

			if(head &&!tail)
				this->append(ptr);
			else
				this->insert(ptr);
		}

		void insert (std::shared_ptr<Node> new_node) 
		{	
			new_node->head = head;
			new_node->tail = tail;

			head->head = new_node;
			if(head->tail == nullptr)
				tail = new_node;

			head = std::move(new_node);
		}

		void append (std::shared_ptr<Node> new_node) 
		{
			new_node->head = head;
			new_node->tail = tail;

			tail->tail = new_node;
			if(tail->head == nullptr)
				head = new_node;

			tail = std::move(new_node);
		}

		void print ()
		{
			std::cout << "[ ";
			std::shared_ptr<Node> ptr = head;
			while(ptr!= nullptr) 
			{	
				std::cout << ptr->data << " ";
				ptr = ptr->tail;
			}
			std::cout << "]"<<std::endl;
		}

		 
		void deleteData(T d)
		{	
			
			if(head == nullptr )
			{
				return;
			}
			if(head->data == d)
			{
				head = head->tail;
			}
			else
			{
					std::shared_ptr<Node> iterator = head;

				while(iterator->tail!= nullptr)
				{
					if(iterator->tail->data == d)
					{
						iterator->tail = iterator->tail->tail;
					}
					else 
					{
						iterator = iterator->tail;
					}
				}
			}
		}
	
		void deleteAll ()

		{	 
			while (head!= nullptr)
			{
				head = head->tail;	
			}
		}
	 


};

int main()
{

	std::cout<<"Hello world"<<std::endl;

	MyList <int> nums;

	for ( int i = -100; i < 300; i += 21)

	{
		nums.add(i);
	}	

	nums.print ();

	nums.deleteData(129);

	nums.print ();

	nums.deleteAll ();

	nums.print ();

	std::cout<<std::endl;

	nums.add(-449);

	nums.print ();

}

template<typename T>
class BinaryTree
{
    struct Node
    {
        T mData{};
        Node* mLchild{};
        Node* mRchild{};
    };

    Node* mRoot{};

public:

    class iterator
    {
        Node* mCurrent{};

    public:
        using difference_type = std::ptrdiff_t;
        using value_t = T;
        using pointer = T*;
        using reference = T&;
        using iterator_category = std::bidirectional_iterator_tag;

        iterator(Node* n = nullptr) : mCurrent{n} {}

        iterator& operator++()
        {
            mCurrent = mCurrent->mLchild;
            return *this;
        }

        iterator operator++(int) {
            iterator retval = *this;
            if (mCurrent)
            {
                mCurrent = mCurrent->mLchild;
            }
            return retval;
        }

        iterator& operator--()
        {
            mCurrent = mCurrent->mLchild;
            return *this;
        }

        iterator operator--(int) {
            iterator retval = *this;
            if (mCurrent)
            {
                mCurrent = mCurrent->mLchild;
            }
            return retval;
        }

        bool operator==(iterator rhs) { return mCurrent == rhs.mCurrent; }
        bool operator!=(iterator rhs) { return!operator==(rhs); }

        pointer operator->() {
            return mCurrent->mLchild;
        }

        reference operator*() {
            return *(mCurrent->mLchild);
        }
    }

     
    void add(const T data) {}

     
    iterator begin() {
        return iterator(mRoot);
    }

    iterator end() {
        return iterator();
    }

private:

     
    void traverse(Node* startingNode) {}

};
