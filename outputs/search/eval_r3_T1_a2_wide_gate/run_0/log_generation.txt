    if(b + a > CHAR_MIN && CHAR_MAX - b + a > 0){

        return INT_MIN;
    }

int mul(int a, int b){
    int l_a, l_b;
    l_a = a > 0? a : -a;
    l_b = b > 0? b : -b;
    bool flag = false;
    if(a == INT_MIN && b == INT_MIN){
        if(add(0, 0) == 0){
            return 1;
        }else{
            return INT_MIN;
        }
    }
    if( l_a - 1 < l_b ) return INT_MIN;
    if( l_a == l_b ) return flag? 1: INT_MIN;
    int ans = 0;
    for(int i = 0; i < l_a; i++){
        for(int j = 0; j < l_b; j++){
            if(ans / CHAR_MAX && ans % CHAR_MAX + i * l_b + j > CHAR_MAX && !flag){
                flag = true;
                continue;
            }
            ans += i * l_b + j;
        }
    }
    if(flag) return INT_MIN;

    return ans;
}
