## C++17

- C++17 changed the way for-loop has to operate.

```cpp
 
#include <iostream>
#include <list>
int main ()
{
  std::clog << 1;
  std::list<int> numbers { 1, 3, 5, 7 };
  for (auto iter = numbers.begin(); iter!= numbers.end(); ++iter)  
    for(int i = *iter; i> 0; --i) {
      if (i % 2 == 0) {
        change(*iter);
      } else {
        *iter += 2;
      }
    }
  for (auto iter = numbers.begin(); iter!= numbers.end(); ++iter)
    std::clog <<  *iter;;
  std::clog << std::endl;
}
void change(int& p)
{
  p += 3;
}
void doubl(int& p)
{
  p *= 2;
}

void quickSort(int *arr, int low, int high){
  int pivot;
  if(low < high){
    pivot = partition(arr, low, high);
    quickSort(arr, low, pivot-1);
    quickSort(arr, pivot+1, high);
  }
}

int partition(int *arr, int low, int high){
  int i=low-1;
  pivot = arr[high];
  for (int j = low; j < high; j++){
    if (arr[j] <= pivot){
      i++;
      if (i > j)
        break;
        int temp = arr[j];
        arr[j] = arr[i];
        arr[i] = temp;
    }
  }
  i++;
  int temp2;
  temp = arr[i];
  arr[i] = pivot;
  pivot = temp;
  return i;
}

int create_child_process(const pthread_t pchild)
{
    int pid = fork();

    if (pid < 0)
    {
        std::cout << __func__ << " -- " << __LINE__ << " --" << std::endl;
        return -1;
    }

    if (pid == 0)
    {
        std::cout << __func__ << " -- " << __LINE__ << " --" << std::endl;
        child_thread("1");
        pthread_exit(NULL);
    }
    else
    {
        std::cout << __func__ << " -- " << __LINE__ << " --" << std::endl;
        return 0;
    }
}

            cin >> A[i];
            max = A[i] > max? A[i] : max;
            min = A[i] < min? A[i] : min;
        }
         
        double average = max + min;
        int i = max;
        i = max > min? max : min;

        mys::cout << endl;
        mys::cout << "Min: " << min;
        mys::cout << endl;
        mys::cout << "Average: " << average / 2.0 << endl;
    }
