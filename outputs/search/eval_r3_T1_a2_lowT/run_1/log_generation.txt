public:
    person_t_ref()
        : m_pPerson(0)
    { }

    person_t_ref(const person_t &other)
        : m_pPerson(0)
    {
        new (this) person_t_ref(other);
    }
    person_t_ref(const person_t_ref &other)
        : m_pPerson(0)
    {
        new (this) person_t_ref(other.m_pPerson->clone());
    }
    ~person_t_ref()
    {
        delete m_pPerson;
    }

    person_t_ref &operator = (const person_t_ref &other)
    {
        person_t *pNewPerson = new person_t(*(other.m_pPerson));
        person_t *pOldPerson = m_pPerson;
        m_pPerson = pNewPerson;
        delete pOldPerson;
        return *this;
    }

    const person_t &ref() const
    {
        return *m_pPerson;
    }

private:
    person_t* clone() const
    {
        return new person_t(*(this->m_pPerson));
    }

    person_t *m_pPerson;
};

{
public:

    {
        cout << "Base default constructed()\n";
    }

    {
        cout << "Base copy constructed()\n";
    }
    Base(int a)
    {
        cout << "Base constructor(\n";
    }
    virtual ~Base()
    {
        cout << "Base destructed()\n";
    }
    void base_virtual()
    {
        cout << "Base::virtual()\n";
    }
    virtual void virtual_func()
    {
        cout << "Base::virtual_func()\n";

#include <iostream>

template<typename S>
class MyStack
{
public:
  void push(const S& aVal)
  {
    S* new_item = new S;
    *new_item = aVal;
    top_item->next = new_item;
    top_item = new_item;
  }
  S& pop()
  {
    delete top_item;
    top_item = top_item->next;
    return *top_item;
  }
  bool empty() { return top_item==0; }
  MyStack() { top_item = 0; }
  ~MyStack() { while (!empty()) pop(); }
private:
  S* top_item;
};

class Data
{
public:
  Data(char * c) { this->c = *c; cout << this->c << endl; }
  char* getChar() { return &c; }
private:
  char c;
};

int main(int argc, char **argv)
{
  Data d("hola");
  MyStack <Data> s;
  s.push(d);
  cout << *s.pop().getChar() << endl;
}
