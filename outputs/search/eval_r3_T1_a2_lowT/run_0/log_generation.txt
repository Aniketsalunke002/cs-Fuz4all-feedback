void nextDay(char *str){
    time_t rawtime;
    struct tm *timeinfo;
    char date[100];
    time (&rawtime);
    timeinfo = localtime (&rawtime);
    timeinfo->tm_sec+=24;
    strftime(date, 100, "%d.%m.%Y", timeinfo);
    strcpy(str,date);
}
