#include <vector>
using namespace std;
int main() {
	vector<int> a=vector<int>();
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		int m;
		cin>>m;
		a.push_back(m);
	}
	
	long sum=0,max=0;
	cout<<"Input: ";
	for(int i=0;i<n;i++) {
		cout<<" "<<a[i];
	}
	
	for(int i=1;i<(int)a.size();i++) {
		
		if(a[i]==1) {
			if(a[i-1]!= a[i] and a[i-1]) {
				sum+=i;
			}
		}
		if(a[i]<0) {
			max-=a[i];
		}
		else if(a[i]>0) {
			max+=a[i];
		}
	}
	
	cout<<"\nSum of all positive number indexes: "<<sum<<endl;
	cout<<"Max difference of + and - elements: "<<max;
	cout<<endl<<endl;
}

    setString("");
    while(scanf("%s",string))
    {
        truncateString(string);
        sortString();
        displayString();

void setString(char *val){
    if(val==NULL){
        return;
    }
    int len=strlen(val);
    int size=(2*len+1);
    string=(char *)malloc(size);
    string=strtok(val," ");
}
void truncateString(char *str){
    while(*str!='\0')
    {
        *str++;
    }
    *str='\0';
}
void sortString(void){
    char buffer[BUFSIZE];
    strcpy(buffer,string);
    strcpy(string,strtok(buffer," "));
}
void displayString(void){
    printf("%s\n",string);
}

class FloatList
{

 public:
    FloatList(std::string initPath = "/tmp/floatlist")
    {
        std::fstream f{initPath, std::ios::in};
        if (initPath == "-")
        {
            f = std::cin;
        }
        int i = 0;
        float fValue;
        float valueSum{0.0f};
        while (f >> fValue)
        {
            valueSum += fValue;
            values[i++] = fValue;
            if (i >= listSize)
            {
                restructureList(i);
            }
        }
        sum = valueSum;
        std::cout << "Read list from " << initPath << " with " << i - 1 << " elements. "
                  << "Sum of elements is " << valueSum << " (" << values.size() * sizeof(float)
                  << " Bytes).\n";
        if (valueSum < 0)
        {
            negativeNumbers = true;
        }
    }
    void print()
    {
        for (const auto &v : values)
        {
            std::cout << v << " ";
        }
        std::cout << "\n"
                  << "Negative numbers? " << negativeNumbers << "\n";
    }
    int size() const
    {
        return values.size() - 1;
    }

 private:
    void restructureList(int newSize)
    {
        int i = listSize - 1;
        std::vector<float> tmpValues{newSize + 1};
        std::reverse_copy(values.begin(), values.end(), tmpValues.begin());
        size_t offset{0};
        for (const auto &v : tmpValues)
        {
            values[i - offset] = v;
            std::advance(i, 1);
            offset++;
        }
    }

    static constexpr size_t listSize{1 << 4};
    std::vector<float> values{listSize};
    float sum;
    bool negativeNumbers{false};
};

The previous programs generate the same code as the original one.
It appears that the compiler has an optimization done somewhere which
replaces all the i + f(i) by i + f(i) + 1. That's because in C++/STL
it is assumed that the return value of an expression is equal to the
input if there is no modification to it by the expression. 
(In C language, the programmer must explicitly use something like
 i = i + f(i) in order to get the input without any changes.)

class Vec
{
    int len;
    T* vec_arr;
    template <typename Ty_other>
    friend std::ostream& operator<<(std::ostream& os, Vec<Ty_other>& vv);
public:
    Vec() {
        vec_arr = nullptr;
        len = 0;
    }
    Vec(int new_len) {
        len = new_len;
        vec_arr = new T[len];
    }
    Vector<T>& operator=(const Vec& rhs) {
        delete[] this->vec_arr;
        this->Len = rhs.Len;
        this->vec_arr = new T[rhs.Len];
        for (int i=0; i < this->Len; i++)
            this->vec_arr[i] = rhs.vec_arr[i];
        return *this;
    }
    ~Vec() {
        delete[] this->vec_arr;
        this->Len = 0;
    }
};

}
