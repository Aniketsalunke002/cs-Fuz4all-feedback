        foo(x - y);  
        s = x + y + z;  
        foo(++n);  
        foo(x++);  
        x = 10;  
        s = (x + 3) / (++y) + 6 * z;  
        if (z > 0)
        {
            z = z + s;
        }
        foo(s = y++);  
        foo(foo(++z++));  
        s = y < z;  
        foo(n -= ++x);  
        y = ++x + 4;  
        n = -3;  
    }
    foo(n);  
}
int main()
{
    std::cout << "Start\n";
    bar();
    std::cout << "End\n";
    return 0;
}

#include <iostream>
template <typename T>
class Vec2d {
public:
	Vec2d(T x=0, T y=0): x(x), y(y) { return; }
	T getX() { return x; }
	T getY() { return y; }
	void setX(T newx) { x = newx; }
	void setY(T newy) { y = newy; }
	Vec2d<T> operator+(const Vec2d<T>& rhs);
	Vec2d<T> operator-(const Vec2d<T>& rhs) {
		int a = x-rhs.getX();
		int b = y-rhs.getY();
		Vec2d<T> result (a,b);
		return result;
	}
	Vec2d<T> operator*(const Vec2d<T>& rhs);
	Vec2d<T> operator/(const Vec2d<T>& rhs);
private:
	T x;
	T y;
};
 

int main () {

  Vec2d<int> b(1, 3);
}
