  while (counter1 <= 10) {
    if (counter2 == 20) {
      vec1.push_back(counter1);
      counter2 = 10;
    }
    else {
      vec1.push_back(counter1);
      counter2++;
    }
    counter1++;
  }
  counter1 = 1;
  counter2 = 10;

  while (counter2 <= 100) {
    if (counter1 == 20) {
      vec2.push_back(counter2);
      counter1 = 10;
    }
    else {
      vec2.push_back(counter2);
      counter1++;
    }
    counter2++;
  }

   
  for (auto index = 0; index < vec1.size(); index++) {
    vec.push_back(vec1[index]);
    vec.push_back(vec2[index]);
  }
  
   
  vec.erase(vec.begin());
  vec.erase(vec.end() - 2);

  std::cout << "Result: ";
  for (const auto& item : vec) {
    std::cout << item << ", ";
  }

struct Point {
    int x, y;
    bool valid;
    Point(int X = 0, int Y = 0, bool valid_ = 1)
    :x(X), y(Y), valid(valid_){}
    friend void print_point(Point const &pt,...) {
        va_list args;
        va_start(args, pt);
        auto out = va_arg(args, FILE *);
        if (out == NULL) {
            out = stdout;
        }
        fprintf(out, "%s %d %d %d\n", 
            (pt.valid? "" : "Invalid "), pt.x, pt.y,
            pt.x * pt.x + pt.y * pt.y);
        va_end(args);
    }
};

    Point const A(1,3);
    Point const B(2,4, 1);
    Point const *L[] = {&A, &B};
    auto* out = fopen("output.txt", "w");
    for (auto&& pt : L) {
        print_point(*pt, out);
    }
    fclose(out);
    return 0;
}

    if (str[index]!= 0) {
        char next = str[index] + 1;
        str[index] = (next > str[index])? 'a' : next;
    }
}

class StateMachine
{
    private:
        states State;
        int Data = 0;
    public:
        void start(int Data)
        {
            this->State = start;
            this->Data = Data;
        }
        void go();
        states getstate()
        {
            return this->State;
        }
};
void StateMachine::go()
{
    if(State == start)
    {
         
        State = ready;
        Data = Data*100;
    }
    else if( State == ready)
    {
         
        State = finish;
    }
}

{
