#include <iostream>
#include <string>
using namespace std;
class Shape {
protected:
  string type;
public:
  void print() const;
  const string& getType() const;
};

inline void Shape::print() const {
  cout << "                      name\n"
       << "                      _____\n"
       << "                   shapetype: "
       << type << endl;
};

inline const string& Shape::getType() const {
  return type;
}

class Rectangle : public Shape {
private:
  int width;
  int length;
public:
  Rectangle(
      string t,
      int w,
      int l
  );
};

Rectangle::Rectangle(
    string t,
    int w,
    int l
) : Shape{t} {
  width = w;
  length = l;
}

int main() {
  Rectangle r("Rectangle", 10, 20);
  r.print();
  cout << r.getType();
  return 0;
}

bool binsearch(int *arr, int length, int value) {
    int l = 0;
    int r = length;
    while (l < r) {
        int m = (r + l) / 2;
        if (arr[m] == value) {
            return true;
        } else if (value > arr[m]) {
            l = m + 1;
        } else {
            r = m;
        }
    }
    return false;
}
